// Imports

var debTkr = import My/Lib/Ranged_Debuffing/public

// ****************************************** //

// Procedure:

// Call every frame
debTkr.track()

?foe.distance > 22
  equipL triskelion
  equipR shield compound *max

// Check if safe to apply debuffs
:?debTkr.canDebuff("debuff_chill")
  equip max* wand dI

:?debTkr.canDebuff("debuff_damage")
  equip max* crossbow dP
:
  equipR repeating crossbow *max

// ****************************************** //

// Monitor for debug
debTkr.monitor()
